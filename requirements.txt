#   1. System tools (macOS / Homebrew):
#        brew install ffmpeg ollama git-lfs
#        brew services start ollama
#        ollama pull qwen2.5:3b      # ~2 GB local LLM weights
#
#   2. TransNetV2 scene-cut model (vendored, NOT a pip package):
#        mkdir -p third_party && cd third_party
#        git clone https://github.com/soCzech/TransNetV2.git
#        cd ..
#
#      The model weights live in Git LFS, but the upstream repo's free LFS
#      bandwidth is frequently exhausted. Pull them via the GitHub media CDN
#      (which uses a separate quota and works reliably):
#        BASE=https://media.githubusercontent.com/media/soCzech/TransNetV2/master/inference/transnetv2-weights
#        DIR=third_party/TransNetV2/inference/transnetv2-weights
#        curl -sL -o "$DIR/saved_model.pb"                            "$BASE/saved_model.pb"
#        curl -sL -o "$DIR/variables/variables.data-00000-of-00001"   "$BASE/variables/variables.data-00000-of-00001"
#        curl -sL -o "$DIR/variables/variables.index"                 "$BASE/variables/variables.index"
#
#      Verify (saved_model.pb should be ~6 MB, SHA256 below):
#        shasum -a 256 "$DIR/saved_model.pb"
#        # expected: 8ac2a52c5719690d512805b6eaf5ce12097c1d8860b3d9de245dcbbc3100f554
